        // Mathematical Constants derived from MARDI & LPNM MD2 growth dynamics
        const L = 1800; // Carrying capacity (Target mature fruit weight in grams)
        const k = 0.35; // Growth rate constant
        const x0 = 12;  // Midpoint of rapid development (weeks post-forcing)

        // Calculate logistic function f(x)
        function f(x) {
            return L / (1 + Math.exp(-k * (x - x0)));
        }

        // Calculate analytical derivative f'(x)
        function fPrime(x) {
            const expTerm = Math.exp(-k * (x - x0));
            const denominator = 1 + expTerm;
            return (L * k * expTerm) / (denominator * denominator);
        }

        // Compute tangent line points for a selected week
        // We calculate points for x from x - 3 to x + 3 to draw a beautiful localized tangent line
        function getTangentDataset(selectedX) {
            const y0 = f(selectedX);
            const slope = fPrime(selectedX);
            
            const points = [];
            const start = Math.max(0, selectedX - 3.5);
            const end = Math.min(24, selectedX + 3.5);
            
            for (let x = start; x <= end; x += 0.5) {
                const y = slope * (x - selectedX) + y0;
                points.push({ x: x, y: y });
            }
            return points;
        }
